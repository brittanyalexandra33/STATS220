Stats Meme

library(magick)
happy_bob <- image_read("https://png.pngitem.com/pimgs/s/587-5871870_rage-comics-troll-face-png-image-meme-face.png") %>% 
  image_scale(300)

scary_bob <- image_read("https://www.kindpng.com/picc/m/390-3901764_rage-comic-faces-sad-hd-png-download.png") %>% 
  image_scale(300)

happy_text <- image_blank(width = 300,
                          height = 300,
                          color = "#000000") %>%
  image_annotate(text = "Taking a Statistics \nDegree because I \nenjoy the Subject",
                 color = "#FFFFFF",
                 size = 30,
                 font = "Impact",
                 gravity = "center")

scary_text <- image_blank(width = 300,
                          height = 300,
                          color = "#000000") %>%
  image_annotate(text = "Having to take\ncoding papers",
                 color = "#FFFFFF",
                 size = 30,
                 font = "Impact",
                 gravity = "center")

first_row <- c(happy_bob, happy_text) %>%
  image_append()

second_row <- c(scary_bob, scary_text) %>%
  image_append()

meme <- c(first_row, second_row) %>%
  image_append(stack = TRUE)

image_write(meme, "Stats220meme.png")

library(magick)

heart_image <-image_read("https://thumbs.dreamstime.com/b/vector-hand-drawn-simple-ink-heart-illustration-love-symbol-decor-element-logos-labels-emblems-posters-postcards-vector-137492806.jpg") %>%
  image_scale(500)

heart_image2 <- image_read("https://i.pinimg.com/474x/17/bd/80/17bd80b2ebc23ae5b3f85f152bc22474.jpg") %>%
  image_scale(500)

heart_image3 <-image_read("https://thumbs.dreamstime.com/b/cute-kawaii-happy-pink-heart-cute-kawaii-happy-pink-heart-vector-illustration-165119288.jpg") %>%
  image_scale(500)

heart_image4 <- image_read("https://static.vecteezy.com/system/resources/previews/004/649/201/original/cute-heart-emoji-free-vector.jpg") %>%
  image_scale(500)

frame1 <-heart_image %>%
  image_annotate(text="You make my heart happy", size = 25, gravity = "north") %>% 
  image_scale(200) %>%
  image_extent("500x500")

frame2 <- heart_image2 %>%
  image_annotate(text="You make my heart happy", size = 30, gravity = "north") %>%
  image_scale(300) %>%
  image_extent("500x500")

frame3 <-heart_image3 %>%
  image_annotate(text="You make my heart happy", size = 35, gravity = "north") %>%
  image_scale(400) %>%
  image_extent("500x500")

frame4 <- heart_image4 %>%
  image_annotate(text ="You make my heart happy", size = 35, gravity = "north") %>%
  image_scale(500) %>%
  image_extent("500x500")

frames <- c(frame1, frame2, frame3, frame4) 

gif = image_animate(frames, fps = 1)
